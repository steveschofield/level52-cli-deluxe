# Install Powershell on Kali Linux
# 1. Update your system first
sudo apt update && sudo apt upgrade -y

# 2. Install required dependencies (if not already present)
sudo apt install -y wget tar libc6 libgcc-s1 libgssapi-krb5-2 libicu72 zlib1g libunwind8 libssl3

# 3. Download the latest stable arm64 tar.gz
# Check https://github.com/PowerShell/PowerShell/releases for the newest version
# Example with 7.5.4 (adjust the version number if newer available)
wget https://github.com/PowerShell/PowerShell/releases/download/v7.5.4/powershell-7.5.4-linux-arm64.tar.gz -O /tmp/powershell.tar.gz

# 4. Create install directory
sudo mkdir -p /opt/microsoft/powershell/7

# 5. Extract the archive
sudo tar zxf /tmp/powershell.tar.gz -C /opt/microsoft/powershell/7

# 6. Set execute permissions
sudo chmod +x /opt/microsoft/powershell/7/pwsh

# 7. Create symlink so you can run 'pwsh' from anywhere
sudo ln -s /opt/microsoft/powershell/7/pwsh /usr/bin/pwsh

# 8. Clean up
rm /tmp/powershell.tar.gz
